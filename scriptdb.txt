-- Table: dpt_estadal

-- DROP TABLE dpt_estadal;

CREATE TABLE dpt_estadal
(
  gid integer NOT NULL DEFAULT nextval('"dpt estadal_gid_seq"'::regclass),
  cod_estado character varying(50),
  estado character varying(20),
  km2 numeric,
  sup_ha numeric,
  geom geometry(MultiPolygon,4189),
  CONSTRAINT "dpt estadal_pkey" PRIMARY KEY (gid)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE dpt_estadal
  OWNER TO postgres;

-- Index: "dpt estadal_geom_gist"

-- DROP INDEX "dpt estadal_geom_gist";

CREATE INDEX "dpt estadal_geom_gist"
  ON dpt_estadal
  USING gist
  (geom);
