(
  cod_dp_enc integer NOT NULL, -- Codigo del encuestado
  reg_ele_sit_pol character(1) NOT NULL, -- Registro electoral, opciones si es S es que si esta inscrito, si es N es por que no esta inscrito
  par_pol_sit_pol character(1), -- Partido politico, opciones S cuando es si y N cuando es no
  cod_par_pol integer, -- Codigo del partido politico
  otr_par_sit_pol character varying(60), -- Si pertenece a otro partido politico
  nom_cen_vot_sit_pol character varying(80), -- Nombre del centro de votacion
  cod_par character(6), -- Codigo de la parroquia
  mie_mes_sit_pol character(1), -- Miembro de mesa, opciones S para cuando es si y N para cuando es no
  tes_mes_sit_pol character(1), -- Testigo de mesa, opciones S para cuando es si y N para cuando es no
  res_par_sit_pol character(1), -- Si tiene responsabilidad en el partido politico, opciones S para cuando es si y N para cuando es no
  ins_sit_pol character varying(60), -- Instancia a la que pertenece
  des_res_sit_pol character varying(60), -- Descripcion de la responsabilidad
  niv_dir_sit_pol character(1), -- Nivel de direccion, opciones N para cuando es nacional, E para cuando es estadal, M para cuando es municipal y P para cuando es parroquial
  CONSTRAINT "PK_situacion_politica" PRIMARY KEY (cod_dp_enc),
  CONSTRAINT "FK_dp_encuestado_a_situacion_politica" FOREIGN KEY (cod_dp_enc)
      REFERENCES viviendo.mvv_dp_encuestado (cod_dp_enc) MATCH SIMPLE
      ON UPDATE RESTRICT ON DELETE RESTRICT,
  CONSTRAINT "FK_partido_politico_a_situacion_politica" FOREIGN KEY (cod_par_pol)
      REFERENCES viviendo.mvv_partido_politico (cod_par_pol) MATCH SIMPLE
      ON UPDATE RESTRICT ON DELETE RESTRICT

WITH (
  OIDS=FALSE
);
